data, label=generate.clusters(3, 75, radii=(0.7, 0.9, 1), dimensions=3, shape="curve",\
    parametric=(lambda a: (2*a-1,2*(2*a-1),(a*2-1)**2), lambda a:((2*a-1)**3-2*(2*a-1), 4*(2*a-1)+1, (2*a-1)**2-1), lambda a:((2*a-1), 2*(2*a-1)-6, -(2*a-1)))\
    )

parametric_t=np.linspace(0, 1, 50)
line1=lambda a: (2*a-1,2*(2*a-1),(a*2-1)**2)
line1=line1(parametric_t)
line2=lambda a:((2*a-1)**3-2*(2*a-1), 4*(2*a-1)+1, (2*a-1)**2-1)
line2=line2(parametric_t)
line3=lambda a:((2*a-1), 2*(2*a-1)-6, -(2*a-1))
line3=line3(parametric_t)

fig=plot.figure(figsize=(10, 10))
graph=fig.add_subplot(projection='3d')
graph.scatter(data[:,0], data[:,1], data[:,2], c=label)

graph.plot(line1[0], line1[1], line1[2], linewidth=3, c="#992299")
graph.plot(line2[0], line2[1], line2[2], linewidth=3, c="#339999")
graph.plot(line3[0], line3[1], line3[2], linewidth=3, c="#cccc44")

graph.view_init(20, 110)
